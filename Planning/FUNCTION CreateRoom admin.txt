Function CreateRoom(jwt_token, room_settings):

//Verify JWT token
user_info = VERIFY_JWT(jwt_token)

IF user_info is invalid:
	RETURN error: "Please login or sign up first before create a room"

// Use user_info to create the room
	room_id = GENERATE_UNIQUE_ID()

	room_data = {
        _id: room_id,
        owner_id: user_info.id,
        room_name: room_settings.name,
        settings: room_settings,
        is_ready: false,               // Default: quiz not ready yet
        participants: [],             // Empty participant list initially
        created_at: CURRENT_TIMESTAMP
    }

	INSERT room_data INTO rooms_collection

RETURN success: "Room created", room_id: room_id
